    <script>
        function verifyLogin() {
            if(document.getElementById('mainPass').value === '1245') {
                document.getElementById('loginOverlay').classList.add('hidden');
                document.getElementById('contentWrapper').classList.remove('hidden');
                document.getElementById('mainBody').classList.remove('overflow-hidden');
            }
        }
        function doCalc() {
            const a = document.getElementById('inAmt').value;
            const t = document.getElementById('inTenor').value;
            if(a > 0) {
                const emi = Math.round((parseInt(a) + (parseInt(a) * 0.105)) / parseInt(t));
                return emi;
            }
            return 0;
        }
        function makeLetter() {
            const emi = doCalc();
            document.getElementById('outName').innerText = document.getElementById('inName').value;
            document.getElementById('outAddr').innerText = document.getElementById('inAddr').value;
            document.getElementById('outAmt').innerText = parseInt(document.getElementById('inAmt').value).toLocaleString('en-IN');
            document.getElementById('outTenor').innerText = document.getElementById('inTenor').value;
            document.getElementById('outEmi').innerText = emi.toLocaleString('en-IN');
            document.getElementById('outPan').innerText = document.getElementById('inPan').value || "NA";
            document.getElementById('outAadhar').innerText = document.getElementById('inAadhar').value || "NA";
            document.getElementById('outBank').innerText = document.getElementById('inBank').value;
            document.getElementById('outAccUser').innerText = document.getElementById('inAcc').value;
            document.getElementById('outIfsc').innerText = document.getElementById('inIfsc').value;
            document.getElementById('outDate').innerText = new Date().toLocaleDateString('en-IN', {day:'2-digit', month:'long', year:'numeric'});
            document.getElementById('outRef').innerText = 'REL-' + Math.floor(10000000 + Math.random() * 90000000);
            document.getElementById('formSection').classList.add('hidden');
            document.getElementById('letterSection').classList.remove('hidden');
            window.scrollTo(0,0);
        }
